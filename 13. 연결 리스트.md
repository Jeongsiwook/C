### Linked list
   ```
   #include <stdio.h>
   #include <string.h>
   #define MAX_POIS 100
   
   typedef struct _Poi {
       char name[100];
       double longitude;
       double latitude;
       struct _Poi *next;
   } Poi;
   
   int main()
   {
      Poi seoul = {"Seoul", 130.0, 37.0, NULL};
      Poi busan = {"Busan", 131.0, 35.0, NULL};
      Poi daegu = {"Daegu", 131.0, 36.0, NULL};
      int i;
      
      seoul.next = &busan;
      
      Poi *ptr;
      for (ptr = &seoul; ptr != NULL; ptr = ptr->next) {
          printf("%s => (%lf, %lf)\n", ptr->name, ptr->longitude, ptr->latitude);
      }
      
      printf("Tour plan changed\n");
      
      seoul.next = &daegu;
      daegu.next = &busan;
      
      for (ptr = &seoul; ptr != NULL; ptr = ptr->next) {
          printf("%s => (%lf, %lf)\n", ptr->name, ptr->longitude, ptr->latitude);
      }
  }
  ```
